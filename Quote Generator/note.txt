// HTML'deki 'quote' ve 'author' elementlerini alır.
const quote = document.getElementById("quote"); // Alıntı metninin yerleştirileceği element
const author = document.getElementById("author"); // Yazar adının yerleştirileceği element

// Rastgele alıntı sağlayan API'nin URL'si
const api_url = "https://api.quotable.io/random";

// Asenkron bir fonksiyon tanımlıyor; belirtilen URL'den alıntı alıyor.
async function getquote(url) {
    // URL'ye GET isteği yapar
    const response = await fetch(url);
    // Gelen yanıtı JSON formatında alır
    var data = await response.json();
    // console.log(data); // Yanıtı konsola yazdırmak için; (geliştirme aşamasında kullanılabilir)
    // Alıntı ve yazar adını HTML elementlerine yerleştirir.
    quote.innerHTML = data.content; // Alıntı metnini yerleştirir
    author.innerHTML = data.author; // Yazar adını yerleştirir
}

// getquote fonksiyonu, belirli bir URL ile çağrılır ve alıntıyı alır
getquote(api_url);

// Tweet atmak için fonksiyon
function tweet() {
    // Twitter'a tweet atma URL'si oluşturur
    window.open("https://x.com/intent/tweet?text=" + quote.innerHTML +
    "  - by " + author.innerHTML, "Tweet Window",
    "width=600, height=300"); // Yeni pencerede açar
}
